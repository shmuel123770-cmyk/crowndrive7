rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    function isImage() {
      return request.resource.contentType.matches('image/.*')
             && request.resource.size < 12 * 1024 * 1024;   // up to 12MB
    }
    function isVideo() {
      return request.resource.contentType.matches('video/.*')
             && request.resource.size < 180 * 1024 * 1024;  // up to 180MB
    }

    // User documents (license / selfie) — private to the owner. Images only.
    match /users/{uid}/{allPaths=**} {
      allow read:  if request.auth != null && request.auth.uid == uid;
      allow write: if request.auth != null && request.auth.uid == uid && isImage();
    }

    // Car listing media — owner uploads photos and an optional short demo video;
    // any signed-in user may read (public listing images are also served via
    // makePublic() URLs). Images or videos.
    match /cars/{uid}/{allPaths=**} {
      allow read:  if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == uid && (isImage() || isVideo());
    }

    // Profile avatars — public read (shown across the site), owner-only write.
    match /avatars/{uid}/{allPaths=**} {
      allow read:  if true;
      allow write: if request.auth != null && request.auth.uid == uid && isImage();
    }

    // Booking media (payment proof, handover photos/videos) — PRIVATE. Direct read is limited to the
    // uploader only; the other party + admins read through short-lived signed URLs issued by a Netlify
    // Function (Admin SDK, which verifies booking participation and bypasses these rules). This closes
    // the leak where any signed-in user could read another booking's payment proofs / handover videos.
    match /bookings/{bookingId}/{folder}/{uid}/{allPaths=**} {
      allow read:  if request.auth != null && request.auth.uid == uid;
      allow write: if request.auth != null && request.auth.uid == uid && (isImage() || isVideo());
    }

    // Note: CrownDrive V2 normally uploads/reads through short-lived signed URLs
    // issued by Netlify Functions (Admin SDK), which bypass these rules. These
    // rules are a defense-in-depth layer that also permits correct, size-limited
    // direct image/video writes scoped to each user's own folder.
  }
}
